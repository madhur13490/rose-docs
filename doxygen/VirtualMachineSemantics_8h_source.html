00407         for (Memory::iterator mi=state.mem.begin(); mi!=state.mem.end(); ++mi) {
00408             if (new_cell.must_alias(*mi)) {
00409                 if ((*mi).clobbered) {
00410                     (*mi).clobbered = false;
00411                     (*mi).data = new_cell.data;
00412                     return new_cell.data;
00413                 } else {
00414                     return (*mi).data;
00415                 }
00416             } else if (new_cell.may_alias(*mi) && (*mi).written) {
00417                 aliased = true;
00418             }
00419         }
00420

00422             /* We didn't find the memory cell in the specified state and it's not aliased to any writes in that state. Therefore
00423              * use the value from the initial memory state (creating it if necessary). */
00424             for (Memory::iterator mi=orig_state.mem.begin(); mi!=orig_state.mem.end(); ++mi) {
00425                 if (new_cell.must_alias(*mi)) {
00426                     ROSE_ASSERT(!(*mi).clobbered);
00427                     ROSE_ASSERT(!(*mi).written);
00428                     state.mem.push_back(*mi);
00429                     std::sort(state.mem.begin(), state.mem.end());

00431                 }
00432             }
00433 
00434             /* Not found in initial state. But if we have a known address and a valid memory map then initialize the original
