00120 {
00126         bool operator() (CFGNode cfgn) const
00127         {
00128                 SgNode *node = cfgn.getNode();
00129 
00130                 //If it is the last node in a function call, keep it
00131                 if (isSgFunctionCallExp(node) && cfgn == node->cfgForEnd())
00132                         return true;
00133 
00134                 //The begin edges of basic blocks are not considered interesting, but we would like to keep them
00135                 //This is so we can propagate reachable defs to the top of a basic block
00136                 if (isSgBasicBlock(node) && cfgn == node->cfgForBeginning())
00137                         return true;
00138 
00139                 if (isSgExprStatement(node))
00140                         return (cfgn == node->cfgForBeginning());
00141 
00142                 if (isSgCommaOpExp(node))
00143                         return (cfgn == node->cfgForBeginning());
00144 
00145                 //Remove all non-interesting nodes
00146                 if (!cfgn.isInteresting())
00147                         return false;
00148 
00149                 //Remove all non-end nodes for initNames
00150                 if (isSgInitializedName(node) && cfgn != node->cfgForEnd())
00151                         return false;
00152 
00153                 //Remove non-beginning nodes for try statements
00154                 if (isSgTryStmt(node) && cfgn != node->cfgForBeginning())
00155                         return false;
00156 
00157                 //Use the last node of binary operators
00158                 if (isSgAndOp(node) || isSgOrOp(node))
00159                 {
00160                         if (cfgn != node->cfgForEnd())
00161                                 return false;
00162                 }
00163 
00164                 //We only want the middle appearance of the teritatry operator - after its conditional expression
00165                 //and before the true and false bodies. This makes it behave as an if statement for data flow
00166                 //purposes
00167                 if (isSgConditionalExp(node))
00168                 {
00169                         return cfgn.getIndex() == 1;
00170                 }
00171 
00172                 return true;
