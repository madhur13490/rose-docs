02649     void processBlock(SgAsmBlock* b) {
02650         const SgAsmStatementPtrList& stmts = b->get_statementList();
02651         if (stmts.empty()) return;
02652         if (!isSgAsmInstruction(stmts[0])) return; /* A block containing functions or something */
02653         size_t i = 0;
02654         while (i < stmts.size()) {
02655             size_t oldI = i;
02656             /* Advance until either i points to a repeated string op or it is just after a hlt or int */
02657             while (i < stmts.size() && !isRepeatedStringOp(stmts[i]) && (i == oldI || !isHltOrInt(stmts[i - 1]))) ++i;
02658             processBlock(stmts, oldI, i);
02659             if (i >= stmts.size()) break;
02660             if (isRepeatedStringOp(stmts[i])) {
02661                 processBlock(stmts, i, i + 1);
02662                 ++i;
02663             }
02664             ROSE_ASSERT(i != oldI);
02665         }
