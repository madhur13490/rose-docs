00869     void erase(const Range &erase_range) {
00870         /* This loop figures out what needs to be removed and calls the elements' removing(), truncate(), etc. methods but does
00871          * not actually erase them from the underlying map yet.  We must not erase them yet because the std::map::erase()
00872          * method doesn't return any iterator.  Instead, we create a list (via an iterator range) of elements that will need to
00873          * be erased from the underlying map.
00874          *
00875          * This loop also creates a list of items that need to be inserted into the underlying map.  Even though the
00876          * std::map::insert() can return an iterator, we can't call it inside the loop because then our erasure iterators will
00877          * become invalid. */
00878         if (erase_range.empty())
00879             return;
00880         Map insertions;
00881         iterator erase_begin=end();
00882         iterator ei;
00883         for (ei=lower_bound(erase_range.first()); ei!=end() && !erase_range.left_of(ei->first); ++ei) {
00884             Range found_range = ei->first;
00885             Value &v = ei->second;
00886             if (erase_range.contains(found_range)) {
00887                 /* Erase entire found range. */
00888                 if (erase_begin==end())
00889                     erase_begin = ei;
00890                 v.removing(found_range);
00891             } else if (erase_range.contained_in(found_range, true/*strict*/)) {
00892                 /* Erase middle of found range. */
00893                 assert(erase_begin==end());
00894                 erase_begin = ei;
00895                 RangePair rt = found_range.split_range_at(erase_range.last()+1);
00896                 insertions[rt.second] = v.split(found_range, rt.second.first());
00897                 RangePair lt = rt.first.split_range_at(erase_range.first());
00898                 v.truncate(rt.first, erase_range.first());
00899                 insertions[lt.first] = v;
00900             } else if (erase_range.begins_after(found_range, true/*strict*/)) {
00901                 /* Erase right part of found range. */
00902                 assert(erase_begin==end());
00903                 erase_begin = ei;
00904                 RangePair halves = found_range.split_range_at(erase_range.first());
00905                 v.truncate(found_range, erase_range.first());
00906                 insertions[halves.first] = v;
00907             } else if (erase_range.ends_before(found_range, true/*strict*/)) {
00908                 /* Erase left part of found range. */
00909                 if (erase_begin==end())
00910                     erase_begin = ei;
00911                 RangePair halves = found_range.split_range_at(erase_range.last()+1);
00912                 insertions[halves.second] = v.split(found_range, halves.second.first());

00914             }
00915         }
